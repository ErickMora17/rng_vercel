8) Add auth helper in WebControlServer:

    def _is_authorized(self, req):
        expected = (self.owner_app.settings.api_token or "").strip()
        if not expected:
            return True
        auth = req.headers.get("Authorization", "")
        if auth.startswith("Bearer ") and auth[7:].strip() == expected:
            return True
        if req.args.get("token", "").strip() == expected:
            return True
        return False

9) Add CORS helper:

    def _corsify(self, response):
        origin = (self.owner_app.settings.allowed_origin or "").strip()
        if origin:
            response.headers["Access-Control-Allow-Origin"] = origin
            response.headers["Vary"] = "Origin"
            response.headers["Access-Control-Allow-Headers"] = "Authorization, Content-Type"
            response.headers["Access-Control-Allow-Methods"] = "GET, POST, OPTIONS"
        return response
